x = df.drop(['charges'], axis = 1)
y = df['charges']
x_train, x_test, y_train, y_test = train_test_split(x, y, test_size=0.2, random_state=0)
exp1 = LinearRegression()
exp1.fit(x_train, y_train)
y_pred = exp1.predict(x_test)
